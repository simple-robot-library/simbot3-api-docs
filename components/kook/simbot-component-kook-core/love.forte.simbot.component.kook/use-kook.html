fun <A : Application> ApplicationBuilder<A>.useKook(usageBuilder: KookComponentUsageBuilder<A>.() -> Unit = {})(source)

通过 KookComponentUsageBuilder 来同时配置使用 Kook 组件 和 Kook botManager。

usage:

simbotApplication(Foo) {
   useKook()
   // 或
   useKook {
       // 配置组件
       component {
           // ...
       }
       // 配置botManager
       botManager {
           // ...
       }
   }
}

simbotApplication(Foo) {
   install(KookComponent) {
       // ...
   }

       // ...

}
